function doGet(e) { var cb = e.parameter.callback; var p = e.parameter; var d = p.data ? JSON.parse(decodeURIComponent(p.data)) : p; var r = handleRequest(d); return cb ? ContentService.createTextOutput(cb+'('+r+')').setMimeType(ContentService.MimeType.JAVASCRIPT) : ContentService.createTextOutput(r).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var d = JSON.parse(e.postData.contents); var r = handleRequest(d); return ContentService.createTextOutput(r).setMimeType(ContentService.MimeType.JSON); } function handleRequest(d) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var accion = d.accion || ''; // LOGIN if (accion === 'login') { var sh = ss.getSheetByName('Usuarios'); if (!sh) return JSON.stringify({ok:false,msg:'Hoja Usuarios no encontrada'}); var rows = sh.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]).trim().toLowerCase() === String(d.usuario).trim().toLowerCase() && String(rows[i][3]).trim() === String(d.password).trim()) { return JSON.stringify({ok:true, usuario:rows[i][0], cargo:rows[i][1], estado:rows[i][2]}); } } return JSON.stringify({ok:false, msg:'Usuario o contraseña incorrectos'}); } // BUSCAR PERSONAL if (accion === 'buscar') { var sheet = ss.getSheetByName(d.empresa); if (!sheet) return JSON.stringify({encontrado:false}); var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]).trim()===String(d.dni).trim() || String(parseInt(data[i][0]))===String(d.dni).trim()) { return JSON.stringify({encontrado:true, dni:data[i][0], ap:data[i][1], am:data[i][2], nombres:data[i][3], nac:data[i][4], ec:data[i][5], dir:data[i][6], dep:data[i][7], prov:data[i][8], dist:data[i][9], tel1:data[i][10], tel2:data[i][11], hijos:data[i][12], nhijos:data[i][13], edu:data[i][14], inst:data[i][15], carrera:data[i][16], egreso:data[i][17], incap:data[i][18], cem:data[i][19], celcem:data[i][20]}); } } return JSON.stringify({encontrado:false}); } // REGISTRAR PERSONAL if (accion === 'registrar') { var reg = ss.getSheetByName('BB.REGISTRO'); if (!reg) { reg = ss.insertSheet('BB.REGISTRO'); reg.appendRow(['DNI','ApellidoPaterno','ApellidoMaterno','Nombres', 'Nacionalidad','Genero','EstadoCivil','Direccion','Departamento','Provincia', 'Distrito','Telefono1','Telefono2','TieneHijos','CantHijos', 'NivelEducativo','Institucion','Carrera','AnoEgreso','Incapacitado', 'Parentesco','ContactoEmergencia','CelularEmergencia', 'FechaRegistro','Empresa','Ruta','Codigo','Grupo','Tipo','RegistradoPor']); } reg.appendRow([d.dni,d.ap,d.am,d.nombres,d.nac,d.genero||'',d.ec,d.dir,d.dep,d.prov, d.dist,d.tel1,d.tel2,d.hijos,d.nhijos,d.edu,d.inst,d.carrera,d.egreso, d.incap,d.parentesco||'',d.cem,d.celcem,d.fecha,d.empresa, d.ruta||'',d.codigo||'',d.sector||'',d.tipo||'',d.registradoPor||'']); return JSON.stringify({ok:true}); } // LISTAR USUARIOS if (accion === 'listarUsuarios') { var sh = ss.getSheetByName('Usuarios'); if (!sh) return JSON.stringify({usuarios:[]}); var rows = sh.getDataRange().getValues(); var lista = []; for (var i = 1; i < rows.length; i++) { if (rows[i][0]) lista.push({usuario:rows[i][0], cargo:rows[i][1], estado:rows[i][2]}); } return JSON.stringify({usuarios:lista}); } // AGREGAR USUARIO if (accion === 'agregarUsuario') { var sh = ss.getSheetByName('Usuarios'); if (!sh) return JSON.stringify({ok:false, msg:'Hoja Usuarios no encontrada'}); sh.appendRow([d.usuario, d.cargo, d.estado, d.password]); return JSON.stringify({ok:true}); } // ELIMINAR USUARIO if (accion === 'eliminarUsuario') { var sh = ss.getSheetByName('Usuarios'); if (!sh) return JSON.stringify({ok:false}); var rows = sh.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]).trim().toLowerCase() === String(d.usuario).trim().toLowerCase()) { sh.deleteRow(i + 1); return JSON.stringify({ok:true}); } } return JSON.stringify({ok:false, msg:'Usuario no encontrado'}); } return JSON.stringify({error:'accion no reconocida'}); }
