function doGet(e) { var cb = e.parameter.callback; var p = e.parameter; var d = p.data ? JSON.parse(decodeURIComponent(p.data)) : p; var r = handleRequest(d); return cb ? ContentService.createTextOutput(cb+'('+r+')').setMimeType(ContentService.MimeType.JAVASCRIPT) : ContentService.createTextOutput(r).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var d = JSON.parse(e.postData.contents); var r = handleRequest(d); return ContentService.createTextOutput(r).setMimeType(ContentService.MimeType.JSON); } function handleRequest(d) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var accion = d.accion || ''; // LOGIN if (accion === 'login') { var sh = ss.getSheetByName('Usuarios'); if (!sh) return JSON.stringify({ok:false,msg:'Hoja Usuarios no encontrada'}); var rows = sh.getDataRange().getValues(); var passHash = hashPassword(String(d.password).trim()); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]).trim().toLowerCase() === String(d.usuario).trim().toLowerCase()) { var almacenada = String(rows[i][3]).trim(); // Acepta hash o (por compatibilidad) texto plano la primera vez if (almacenada === passHash || almacenada === String(d.password).trim()) { return JSON.stringify({ok:true, usuario:rows[i][0], cargo:rows[i][1], estado:rows[i][2]}); } return JSON.stringify({ok:false, msg:'Usuario o contraseña incorrectos'}); } } return JSON.stringify({ok:false, msg:'Usuario o contraseña incorrectos'}); } // BUSCAR PERSONAL if (accion === 'buscar') { var sheet = ss.getSheetByName(d.empresa); if (!sheet) return JSON.stringify({encontrado:false}); // TextFinder: busca rápido el DNI en la columna A sin recorrer todo var dniBuscar = String(d.dni).trim(); var col = sheet.getRange(1, 1, sheet.getLastRow(), 1); var encontrado = col.createTextFinder(dniBuscar).matchEntireCell(true).findNext(); if (!encontrado) { // Intentar como número (por si el DNI está sin ceros) encontrado = col.createTextFinder(String(parseInt(dniBuscar))).matchEntireCell(true).findNext(); } if (!encontrado) return JSON.stringify({encontrado:false}); var fila = encontrado.getRow(); var v = sheet.getRange(fila, 1, 1, 21).getValues()[0]; return JSON.stringify({encontrado:true, dni:v[0], ap:v[1], am:v[2], nombres:v[3], nac:v[4], ec:v[5], dir:v[6], dep:v[7], prov:v[8], dist:v[9], tel1:v[10], tel2:v[11], hijos:v[12], nhijos:v[13], edu:v[14], inst:v[15], carrera:v[16], egreso:v[17], incap:v[18], cem:v[19], celcem:v[20]}); } // BUSCAR EN AMBAS HOJAS (Rapel y Verfrut) en una sola llamada — más rápido if (accion === 'buscarAmbas') { var dniB = String(d.dni).trim(); var empresas = ['Rapel','Verfrut']; for (var e = 0; e < empresas.length; e++) { var sh = ss.getSheetByName(empresas[e]); if (!sh) continue; var c = sh.getRange(1, 1, sh.getLastRow(), 1); var enc = c.createTextFinder(dniB).matchEntireCell(true).findNext(); if (!enc) enc = c.createTextFinder(String(parseInt(dniB))).matchEntireCell(true).findNext(); if (enc) { var f = enc.getRow(); var vv = sh.getRange(f, 1, 1, 21).getValues()[0]; return JSON.stringify({encontrado:true, _empresa:empresas[e], dni:vv[0], ap:vv[1], am:vv[2], nombres:vv[3], nac:vv[4], ec:vv[5], dir:vv[6], dep:vv[7], prov:vv[8], dist:vv[9], tel1:vv[10], tel2:vv[11], hijos:vv[12], nhijos:vv[13], edu:vv[14], inst:vv[15], carrera:vv[16], egreso:vv[17], incap:vv[18], cem:vv[19], celcem:vv[20]}); } } return JSON.stringify({encontrado:false}); } // REGISTRAR PERSONAL if (accion === 'registrar') { var reg = ss.getSheetByName('BB.REGISTRO'); if (!reg) { reg = ss.insertSheet('BB.REGISTRO'); reg.appendRow(['DNI','ApellidoPaterno','ApellidoMaterno','Nombres','SigueMismaDir', 'DireccionActual','Departamento','Provincia','Distrito','Telefono1','TieneHijos', 'NivelEducativo','Institucion','Carrera','AnoEgreso','Discapacidad','Parentesco', 'ContactoReferencia','TelefonoReferencia','FechaRegistro','Empresa','Ruta', 'Codigo','Grupo','Tipo','RegistradoPor','EsTardio']); } var parentescoFinal = (d.parentesco === 'Otros' && d.parentescoOtro) ? d.parentescoOtro : (d.parentesco||''); reg.appendRow([d.dni,d.ap,d.am,d.nombres,d.mismadir||'',d.dir,d.dep,d.prov, d.dist,d.tel1,d.hijos,d.edu,d.inst,d.carrera,d.egreso,d.incap,parentescoFinal, d.cem,d.celcem,d.fecha,d.empresaOrigen||'',d.ruta||'',d.codigo||'',d.sector||'', d.tipo||'',d.registradoPor||'',d.tardio?'SI':'NO']); return JSON.stringify({ok:true}); } // BUSCAR UN REGISTRO EN BB.REGISTRO POR DNI (para editar) if (accion === 'buscarBBRegistro') { var bsh = ss.getSheetByName('BB.REGISTRO'); if (!bsh) return JSON.stringify({encontrado:false}); var dniB = String(d.dni).trim(); var col = bsh.getRange(1, 1, bsh.getLastRow(), 1); var enc = col.createTextFinder(dniB).matchEntireCell(true).findNext(); if (!enc) enc = col.createTextFinder(String(parseInt(dniB))).matchEntireCell(true).findNext(); if (!enc) return JSON.stringify({encontrado:false}); var f = enc.getRow(); var v = bsh.getRange(f, 1, 1, 27).getValues()[0]; return JSON.stringify({encontrado:true, fila:f, dni:v[0], ap:v[1], am:v[2], nombres:v[3], mismadir:v[4], dir:v[5], dep:v[6], prov:v[7], dist:v[8], tel1:v[9], hijos:v[10], edu:v[11], inst:v[12], carrera:v[13], egreso:v[14], incap:v[15], parentesco:v[16], cem:v[17], celcem:v[18], fecha:v[19], empresaOrigen:v[20], ruta:v[21], codigo:v[22], sector:v[23], tipo:v[24], registradoPor:v[25], tardio:v[26]}); } // ACTUALIZAR REGISTRO EXISTENTE EN BB.REGISTRO if (accion === 'actualizarRegistro') { var bsh = ss.getSheetByName('BB.REGISTRO'); if (!bsh) return JSON.stringify({ok:false, msg:'Hoja no encontrada'}); var fila = parseInt(d.fila); if (!fila || fila < 2) return JSON.stringify({ok:false, msg:'Fila inválida'}); // Guardar valores anteriores para auditoría var valoresAntes = bsh.getRange(fila, 1, 1, 27).getValues()[0]; var pf = (d.parentesco === 'Otros' && d.parentescoOtro) ? d.parentescoOtro : (d.parentesco||''); bsh.getRange(fila, 1, 1, 27).setValues([[d.dni,d.ap,d.am,d.nombres,d.mismadir||'', d.dir,d.dep,d.prov,d.dist,d.tel1,d.hijos,d.edu,d.inst,d.carrera,d.egreso, d.incap,pf,d.cem,d.celcem,d.fecha,d.empresaOrigen||'',d.ruta||'',d.codigo||'', d.sector||'',d.tipo||'',d.registradoPor||'',d.tardio||'NO']]); // Registrar en AUDITORIA var aud = ss.getSheetByName('AUDITORIA'); if (!aud) { aud = ss.insertSheet('AUDITORIA'); aud.appendRow(['FechaHora','Usuario','Accion','DNI','Fila','CamposModificados','ValoresAntes']); } var cambios = []; var nuevos = [d.dni,d.ap,d.am,d.nombres,d.mismadir||'',d.dir,d.dep,d.prov,d.dist, d.tel1,d.hijos,d.edu,d.inst,d.carrera,d.egreso,d.incap,pf,d.cem,d.celcem,d.fecha, d.empresaOrigen||'',d.ruta||'',d.codigo||'',d.sector||'',d.tipo||'',d.registradoPor||'',d.tardio||'NO']; var nombres = ['DNI','ApPat','ApMat','Nombres','MismaDir','Dir','Dep','Prov','Dist', 'Tel1','Hijos','Edu','Inst','Carrera','Egreso','Incap','Parentesco','Cem','CelCem','Fecha', 'Empresa','Ruta','Codigo','Grupo','Tipo','RegPor','Tardio']; for (var k = 0; k < nuevos.length; k++) { if (String(valoresAntes[k]||'') !== String(nuevos[k]||'')) { cambios.push(nombres[k]+': "'+(valoresAntes[k]||'')+'" → "'+(nuevos[k]||'')+'"'); } } aud.appendRow([new Date(), d.usuarioEditor||'-', 'EDITAR', d.dni, fila, cambios.length+' campo(s)', cambios.join(' | ')]); return JSON.stringify({ok:true, cambios:cambios.length}); } // LISTAR TODOS LOS REGISTROS DE BB.REGISTRO (admin) if (accion === 'listarBBRegistro') { var rsh = ss.getSheetByName('BB.REGISTRO'); if (!rsh) return JSON.stringify({registros:[]}); var rd = rsh.getDataRange().getValues(); var rl = []; for (var i = 1; i < rd.length; i++) { if (!rd[i][0]) continue; rl.push({dni:rd[i][0], ap:rd[i][1], am:rd[i][2], nombres:rd[i][3], fecha:rd[i][19], ruta:rd[i][21], codigo:rd[i][22], sector:rd[i][23], tipo:rd[i][24], registradoPor:rd[i][25], tardio:rd[i][26]}); } return JSON.stringify({registros:rl}); } // CONTAR REGISTROS POR GRUPO (contador compartido) if (accion === 'contarGrupo') { var rsh = ss.getSheetByName('BB.REGISTRO'); if (!rsh) return JSON.stringify({count:0}); var rd = rsh.getDataRange().getValues(); var count = 0; for (var i = 1; i < rd.length; i++) { if (String(rd[i][23]).trim() === String(d.grupo).trim() && String(rd[i][22]).trim() === String(d.codigo).trim()) count++; } return JSON.stringify({count:count}); } // REGISTRO VIA QR (hoja PRE-REGISTRO-QR) if (accion === 'registrarQR') { var qsh = ss.getSheetByName('PRE-REGISTRO-QR'); if (!qsh) { qsh = ss.insertSheet('PRE-REGISTRO-QR'); qsh.appendRow(['DNI','ApellidoPaterno','ApellidoMaterno','Nombres','TieneHijos', 'DireccionActual','Departamento','Provincia','Distrito', 'Telefono','NivelEducativo','Institucion','Carrera','AnoEgreso','Discapacidad', 'Parentesco','ContactoReferencia','TelefonoReferencia','FechaRegistro']); } qsh.appendRow([d.dni, d.ap, d.am, d.nombres, d.hijos||'', d.dir||'', d.dep||'', d.prov||'', d.dist||'', d.tel1||'', d.edu||'', d.inst||'', d.carrera||'', d.egreso||'', d.incap||'', d.parentesco||'', d.cem||'', d.celcem||'', d.fecha]); return JSON.stringify({ok:true}); } // LISTAR USUARIOS if (accion === 'listarUsuarios') { var sh = ss.getSheetByName('Usuarios'); if (!sh) return JSON.stringify({usuarios:[]}); var rows = sh.getDataRange().getValues(); var lista = []; for (var i = 1; i < rows.length; i++) { if (rows[i][0]) lista.push({usuario:rows[i][0], cargo:rows[i][1], estado:rows[i][2]}); } return JSON.stringify({usuarios:lista}); } // AGREGAR USUARIO if (accion === 'agregarUsuario') { var sh = ss.getSheetByName('Usuarios'); if (!sh) return JSON.stringify({ok:false, msg:'Hoja Usuarios no encontrada'}); sh.appendRow([d.usuario, d.cargo, d.estado, hashPassword(String(d.password).trim())]); return JSON.stringify({ok:true}); } // ELIMINAR USUARIO if (accion === 'eliminarUsuario') { var sh = ss.getSheetByName('Usuarios'); if (!sh) return JSON.stringify({ok:false}); var rows = sh.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]).trim().toLowerCase() === String(d.usuario).trim().toLowerCase()) { sh.deleteRow(i + 1); return JSON.stringify({ok:true}); } } return JSON.stringify({ok:false, msg:'Usuario no encontrado'}); } return JSON.stringify({error:'accion no reconocida'}); } // Convierte una contraseña a hash SHA-256 (irreversible) function hashPassword(texto) { var raw = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, texto, Utilities.Charset.UTF_8); var hash = ''; for (var i = 0; i < raw.length; i++) { var b = raw[i]; if (b < 0) b += 256; var h = b.toString(16); if (h.length === 1) h = '0' + h; hash += h; } return hash; } // EJECUTA ESTA FUNCIÓN UNA SOLA VEZ para convertir las contraseñas existentes a hash. // En el editor de Apps Script: selecciona "migrarPasswords" arriba y presiona Ejecutar. function migrarPasswords() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName('Usuarios'); var rows = sh.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { var pass = String(rows[i][3]).trim(); // Si ya es hash (64 caracteres hex) no la toca if (pass.length === 64 && /^[0-9a-f]+$/.test(pass)) continue; if (pass) { sh.getRange(i + 1, 4).setValue(hashPassword(pass)); } } }
